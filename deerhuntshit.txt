#!/bin/sh

DIR=$(pwd)
NAME=$1

if [ $# -lt 1 ]; then
	echo "USAGE: gbc <Input File>"
	exit 1
fi

cp $1 ~/ybc/ 2> /dev/null
cd ~/ybc
./ybc $1
mv a.out "$DIR/$(echo $NAME | sed 's/.b/.out/g')" 2> /dev/null
rm *.b 2> /dev/null
rm a.out 2> /dev/null
